568                  'RHN Certificate appears to have expired: %s' % just_date) 
569              return 11

572          prepRhnCert(options) 
573          # local activation 
574          try: 
575

577              db_host = CFG.DB_HOST

584                          username=db_user, password=db_password, database=database) 
585              if options.verbose: 
586                  print ("Database connectioned initialized: refer to %s" %  
587                         CFG.filename) 
588              activateSatellite_local(options) 
589          except RHNCertLocalActivationException, e: 
590              writeError(e) 
591              return 30

593          # remote activation 
594          if not options.disconnected: 
595              try: 
596                  activateSatellite_remote(options) 
597              except RHNCertRemoteActivationException, e: 
598                  writeError(e) 
599                  return 20 
600              except RHNCertRemoteNoManagementSlotsException, e: 
601                  writeError(e) 
602                  return 80 
603              except RHNCertRemoteSatelliteAlreadyActivatedException, e:

605                  # so let's return 0 
606                  return 0

608                  writeError(e)

616              except RHNCertRemoteSatelliteNotActivatedException, e:

620                  writeError(e) 
621                  return 86 
622              except RHNCertNoSatChanForVersion, e: 
623                  writeError(e) 
624                  return 87

631              try:

638              # already know about.
