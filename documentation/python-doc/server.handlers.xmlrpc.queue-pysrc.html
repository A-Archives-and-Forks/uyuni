644                  and sc.server_id = :server_id 
645                  UNION 
646                  -- when no EVR requested, we need to compute the max available 
647                  -- from the channels the server is subscribed to 
648                  select 
649                      pn.name name, 
650                      max(pevr.evr) evr 
651                  from 
652                      rhnActionPackage ap, 
653                      rhnServerChannel sc, 
654                      rhnChannelPackage cp, 
655                      rhnPackage p, 
656                      rhnPackageEVR pevr, 
657                      rhnPackageName pn 
658                  where 
659                      ap.action_id = :action_id 
660                  and ap.evr_id is null 
661                  and ap.name_id = pn.id 
662                  and ap.name_id = p.name_id 
663                  and p.evr_id = pevr.id 
664                  and sc.server_id = :server_id 
665                  and sc.channel_id = cp.channel_id 
666                  and cp.package_id = p.id 
667                  group by pn.name 
668              ) pl 
669              group by pl.name 
670          ) pkglist 
671          """ 
672          h = rhnSQL.prepare(statement) 
673          h.execute(action_id = actionId, server_id = self.server_id) 
674          ret = h.fetchall_dict() or [] 
675          packages = [] 
676          for p in ret: 
677              # old clients have issues dealing with real epochs, so we 
678              # kind of fake it for now in here 
679              entry = [p['name'], p['version'], p['release'], ''] 
680              packages.append(entry) 
681          xml = xmlrpclib.dumps((packages,), methodname='client.update_packages') 
682          return xml
