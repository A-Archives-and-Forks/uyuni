   168:    function can_entitle_server (
   169:         server_id_in in number,
   170:         type_label_in in varchar2 )
   171:    return number is
   172:       cursor addon_servergroups (base_label_in in varchar2,
   173:                                  addon_label_in in varchar2) is
   174:          select
   175:             addon_id
   176:          from
   177:             rhnSGTypeBaseAddonCompat
   178:          where base_id = lookup_sg_type (base_label_in)
   179:            and addon_id = lookup_sg_type (addon_label_in);
   180: 
   181:       previous_ent        rhn_entitlements.ents_array;
   182:       is_base_in          char   := 'N';
   183:       is_base_current     char   := 'N';
   184:       i                   number := 0;
   185:       sgid                number := 0;
   186: 
   187:    begin
   188: 
   189:       previous_ent := rhn_entitlements.ents_array();
   190:       previous_ent := rhn_entitlements.get_server_entitlement(server_id_in);
   191: 
   192:       select distinct is_base
   193:       into is_base_in
   194:       from rhnServerGroupType
   195:       where label = type_label_in;
   196: 
   197:       if previous_ent.count = 0 then
   198:          sgid := find_compatible_sg (server_id_in, type_label_in);
   199:          if (is_base_in = 'Y' and sgid is not null) then
   200:             -- rhn_server.insert_into_servergroup (server_id_in, sgid);
   201:             return 1;
   202:          else
   203:             -- rhn_exception.raise_exception ('invalid_base_entitlement');
   204:             return 0;
   205:          end if;
   206: 
   207:       -- there are previous ents, first make sure we're not trying to entitle a base ent
   208:       elsif is_base_in = 'Y' then
   209:          -- rhn_exception.raise_exception ('invalid_addon_entitlement');
   210:          return 0;
   211: 
   212:       -- it must be an addon, so proceed with the entitlement
   213:       else
   214: 
   215:          -- find the servers base ent
   216:          is_base_current := 'N';
   217:          i := 0;
   218:          while is_base_current = 'N' and i <= previous_ent.count
   219:          loop
   220:             i := i + 1;
   221:             select is_base
   222:             into is_base_current
   223:             from rhnServerGroupType
   224:             where label = previous_ent(i);
   225:          end loop;
   226: 
   227:          -- never found a base ent, that would be strange
   228:          if is_base_current  = 'N' then
   229:             -- rhn_exception.raise_exception ('invalid_base_entitlement');
   230:             return 0;
   231:          end if;
   232: 
   233:          -- this for loop verifies the validity of the addon path
   234:          for addon_servergroup in addon_servergroups  (previous_ent(i), type_label_in) loop
   235:             -- find an appropriate sgid for the addon and entitle the server
   236:             sgid := find_compatible_sg (server_id_in, type_label_in);
   237:             if sgid is not null then
   238:                -- rhn_server.insert_into_servergroup (server_id_in, sgid);
   239:                return 1;
   240:             else
   241:                -- rhn_exception.raise_exception ('invalid_addon_entitlement');
   242:                return 0;
   243:             end if;
   244:          end loop;
   245: 
   246:       end if;
   247: 
   248:       return 0;
   249: 
   250:    end can_entitle_server;

   316:             insert into rhnServerHistory ( id, server_id, summary, details )

   406:            rhn_entitlements.repoll_virt_guest_entitlements(server_id_in);

   408:       end if;

   471:     -- *******************************************************************
   472:     -- PROCEDURE: repoll_virt_guest_entitlements
   473:     --
   474:     --   Whenever we add/remove a virtualization_host* entitlement from
   475:     --   a host, we can call this procedure to update what type of slots
   476:     --   the guests are consuming.
   477:     --
   478:     --   If you're removing the entitlement, it's
   479:     --   possible the guests will become unentitled if you don't have enough
   480:     --   physical slots to cover them.
   481:     --
   482:     --   If you're adding the entitlement, you end up freeing up physical
   483:     --   slots for other systems.
   484:     --
   485:     -- *******************************************************************
   486:     procedure repoll_virt_guest_entitlements(server_id_in in number)
   487:     is
   488: 
   489:         -- All channel families associated with the guests of server_id_in
   490:         cursor families is
   491:             select distinct cfs.channel_family_id
   492:             from
   493:                 rhnChannelFamilyServers cfs,
   494:                 rhnVirtualInstance vi
   495:             where
   496:                 vi.host_system_id = server_id_in
   497:                 and vi.virtual_system_id = cfs.server_id;
   498: 
   499:         -- All of server group types associated with the guests of
   500:         -- server_id_in
   501:         cursor group_types is
   502:             select distinct sg.group_type, sgt.label
   503:             from
   504:                 rhnServerGroupType sgt,
   505:                 rhnServerGroup sg,
   506:                 rhnServerGroupMembers sgm,
   507:                 rhnVirtualInstance vi
   508:             where
   509:                 vi.host_system_id = server_id_in
   510:                 and vi.virtual_system_id = sgm.server_id
   511:                 and sgm.server_group_id = sg.id
   512:                 and sg.group_type = sgt.id;
   513: 
   514:         -- Virtual servers from a certain family belonging to a speicifc
   515:         -- host that are consuming physical channel slots over the limit.
   516:         cursor virt_servers_cfam(family_id_in in number, quantity_in in number)  is
   517:             select virtual_system_id
   518:             from (
   519:                 select rownum, vi.virtual_system_id
   520:                 from
   521:                     rhnChannelFamilyMembers cfm,
   522:                     rhnServerChannel sc,
   523:                     rhnVirtualInstance vi
   524:                 where
   525:                     vi.host_system_id = server_id_in
   526:                     and vi.virtual_system_id = sc.server_id
   527:                     and sc.channel_id = cfm.channel_id
   528:                     and cfm.channel_family_id = family_id_in
   529:                 order by sc.modified desc
   530:                 )
   531:             where rownum <= quantity_in;
   532: 
   533:         -- Virtual servers from a certain family belonging to a speicifc
   534:         -- host that are consuming physical system slots over the limit.
   535:         cursor virt_servers_sgt(group_type_in in number, quantity_in in number)  is
   536:             select virtual_system_id
   537:             from (
   538:                 select rownum, vi.virtual_system_id
   539:                 from
   540:                     rhnServerGroup sg,
   541:                     rhnServerGroupMembers sgm,
   542:                     rhnVirtualInstance vi
   543:                 where
   544:                     vi.host_system_id = server_id_in
   545:                     and vi.virtual_system_id = sgm.server_id
   546:                     and sgm.server_group_id = sg.id
   547:                     and sg.group_type = group_type_in
   548:                 order by sgm.modified desc
   549:                 )
   550:             where rownum <= quantity_in;
   551:         -- Get the orgs of Virtual guests
   552:         -- Since they may belong to different orgs
   553:         cursor virt_guest_orgs  is
   554:                 select  distinct (s.org_id)
   555:                 from rhnServer s
   556:                     inner join  rhnVirtualInstance vi on vi.virtual_system_id = s.id
   557:                 where
   558:                     vi.host_system_id = server_id_in
   559:                     and s.org_id <> (select s1.org_id from rhnServer s1 where s1.id = vi.host_system_id) ;
   560: 
   561: 
   562:         org_id_val number;
   563:         max_members_val number;
   564:         max_flex_val number;
   565:         current_members_calc number;
   566:         sg_id number;
   567:         is_virt number := 0;
   568:     begin
   569:         begin
   570:           select 1 into is_virt
   571:                 from rhnServerEntitlementView
   572:            where server_id = server_id_in
   573:                  and label in ('virtualization_host', 'virtualization_host_platform');
   574:            exception
   575:                 when no_data_found then
   576:                   is_virt := 0;
   577:         end;
   578: 
   579:         select org_id
   580:         into org_id_val
   581:         from rhnServer
   582:         where id = server_id_in;
   583: 
   584:         -- deal w/ channel entitlements first ...
   585:         for family in families loop
   586:             if is_virt = 0 then
   587:             -- if the host_server does not have virt
   588:             --- find all possible flex slots
   589:             -- and set each of the flex eligible guests to Y
   590:                 UPDATE rhnServerChannel sc set sc.is_fve = 'Y'
   591:                 where sc.server_id in (
   592:                        select virtual_system_id from (
   593:                             select rownum, vi.virtual_system_id,  sfc.max_members - sfc.current_members as free_slots
   594:                             from rhnServerFveCapable sfc
   595:                                 inner join rhnVirtualInstance vi on vi.virtual_system_id = sfc.server_id
   596:                             where vi.host_system_id = server_id_in
   597:                                   and sfc.channel_family_id = family.channel_family_id
   598:                               order by vi.modified desc
   599:                           )
   600:                         where rownum <=  free_slots
   601:                 );
   602:             else
   603:             -- if the host_server has virt
   604:             -- set all its flex guests to N
   605:                 UPDATE rhnServerChannel sc set sc.is_fve = 'N'
   606:                 where
   607:                     sc.channel_id in (select cfm.channel_id from rhnChannelFamilyMembers cfm
   608:                                       where cfm.CHANNEL_FAMILY_ID = family.channel_family_id)
   609:                     and sc.is_fve = 'Y'
   610:                     and sc.server_id in
   611:                             (select vi.virtual_system_id  from rhnVirtualInstance vi
   612:                                     where vi.host_system_id = server_id_in);
   613:             end if;
   614: 
   615:             -- get the current (physical) members of the family
   616:             current_members_calc :=
   617:                 rhn_channel.channel_family_current_members(family.channel_family_id,
   618:                                                            org_id_val); -- fixed transposed args
   619: 
   620:             begin
   621: 
   622:             -- get the max members of the family
   623:             select max_members
   624:             into max_members_val
   625:             from rhnPrivateChannelFamily
   626:             where channel_family_id = family.channel_family_id
   627:             and org_id = org_id_val;
   628: 
   629:             select fve_max_members
   630:             into max_flex_val
   631:             from rhnPrivateChannelFamily
   632:             where channel_family_id = family.channel_family_id
   633:             and org_id = org_id_val;
   634: 
   635:             if current_members_calc > max_members_val then
   636:                 -- A virtualization_host* ent must have been removed, so we'll
   637:                 -- unsubscribe guests from the host first.
   638: 
   639:                 -- hm, i don't think max_members - current_members_calc yielding a negative number
   640:                 -- will work w/ rownum, swaping 'em in the body of this if...
   641:                 for virt_server in virt_servers_cfam(family.channel_family_id,
   642:                                 current_members_calc - max_members_val) loop
   643: 
   644:                     rhn_channel.unsubscribe_server_from_family(
   645:                                 virt_server.virtual_system_id,
   646:                                 family.channel_family_id);
   647:                 end loop;
   648: 
   649:                 -- if we're still over the limit, which would be odd,
   650:                 -- just prune the group to max_members
   651:                 --
   652:                 -- er... wouldn't we actually have to refresh the values of
   653:                 -- current_members_calc and max_members_val to actually ever
   654:                 -- *skip this??
   655:                 if current_members_calc > max_members_val then
   656:                     -- argh, transposed again?!
   657:                     set_family_count(org_id_val,
   658:                                      family.channel_family_id,
   659:                                      max_members_val, max_flex_val);
   660:                     --TODO calculate this correctly
   661:                 end if;
   662: 
   663:            end if;
   664: 
   665:             -- update current_members for the family.  This will set the value
   666:             -- to reflect adding/removing the entitlement.
   667:             --
   668:             -- what's the difference of doing this vs the unavoidable set_family_count above?
   669:             rhn_channel.update_family_counts(family.channel_family_id,
   670:                                              org_id_val);
   671: 
   672:             exception when no_data_found then null;
   673:             end;
   674: 
   675:             -- It is possible that the guests belong  to a different org than the host
   676:             -- so we are going to update the family counts in the guests orgs also
   677:             for org in virt_guest_orgs loop
   678:                     rhn_channel.update_family_counts(family.channel_family_id,
   679:                                              org.org_id);
   680:             end loop;
   681:         end loop;
   682: 
   683:         for a_group_type in group_types loop
   684:           -- get the current *physical* members of the system entitlement type for the org...
   685:           --
   686:           -- unlike channel families, it appears the standard rhnServerGroup.max_members represents
   687:           -- *physical* slots, vs physical+virt ... boy that's confusing...
   688: 
   689:           select max_members, id
   690:             into max_members_val, sg_id
   691:             from rhnServerGroup
   692:             where group_type = a_group_type.group_type
   693:             and org_id = org_id_val;
   694: 
   695: 
   696:       select count(sep.server_id) into current_members_calc
   697:             from rhnServerEntitlementPhysical sep
   698:            where sep.server_group_id = sg_id
   699:              and sep.server_group_type_id = a_group_type.group_type;
   700: 
   701:           if current_members_calc > max_members_val then
   702:             -- A virtualization_host* ent must have been removed, and we're over the limit, so unsubscribe guests
   703:             for virt_server in virt_servers_sgt(a_group_type.group_type,
   704:                                                 current_members_calc - max_members_val) loop
   705:               rhn_entitlements.remove_server_entitlement(virt_server.virtual_system_id, a_group_type.label);
   706: 
   707:               -- decrement current_members_calc, we'll use it to reset current_members for the group at the end...
   708:               current_members_calc := current_members_calc - 1;
   709:             end loop;
   710: 
   711:           end if;
   712: 
   713:           update rhnServerGroup set current_members = current_members_calc
   714:            where org_id = org_id_val
   715:              and group_type = a_group_type.group_type;
   716: 
   717:           -- I think that's all the house-keeping we have to do...
   718:         end loop;
   719: 
   720:     end repoll_virt_guest_entitlements;

   959:     -- *******************************************************************
   960:     -- PROCEDURE: prune_group
   961:     -- Unsubscribes servers consuming physical slots that over the org's
   962:     --   limit.
   963:     -- Called by: set_server_group_count, repoll_virt_guest_entitlements
   964:     -- *******************************************************************
   965:     procedure prune_group (
   966:         group_id_in in number,
   967:         quantity_in in number,
   968:                 update_family_countsYN in number := 1
   969:     ) is
   970:         cursor servergroups is
   971:            select  server_id, server_group_id, sgt.id as group_type_id, sgt.label
   972:             from    rhnServerGroupType              sgt,
   973:                             rhnServerGroup                  sg,
   974:                             rhnServerGroupMembers   sgm
   975:             where   1=1
   976:                     and sgm.server_group_id = group_id_in
   977:                     and sgm.server_id in (
   978:                             select  server_id
   979:                             from    (
   980:                                     select  rownum row_number,
   981:                                                     server_id,
   982:                                                     time
   983:                                     from    (
   984:                                             select  sep.server_id,
   985:                                                     sep.modified time
   986:                                             from
   987:                                                 rhnServerEntitlementPhysical sep
   988:                                             where
   989:                                                 sep.server_group_id = group_id_in
   990:                                             order by time asc
   991:                                     )
   992:                             )
   993:                             where   row_number > quantity_in
   994:                     )
   995:                     and sgm.server_group_id = sg.id
   996:                     and sg.group_type = sgt.id;
   997:       type_is_base char;
   998:     begin
   999:         update        rhnServerGroup
  1000:             set        max_members = quantity_in
  1001:             where    id = group_id_in;
  1002: 
  1003:         for sg in servergroups loop
  1004:             remove_server_entitlement(sg.server_id, sg.label);
  1005: 
  1006:             select is_base
  1007:             into type_is_base
  1008:             from rhnServerGroupType sgt
  1009:             where sgt.id = sg.group_type_id;
  1010: 
  1011:             -- if we're removing a base ent, then be sure to
  1012:             -- remove the server's channel subscriptions.
  1013:             if ( type_is_base = 'Y' ) then
  1014:                    rhn_channel.clear_subscriptions(sg.server_id,
  1015:                                         update_family_countsYN => update_family_countsYN);
  1016:             end if;
  1017: 
  1018:         end loop;
  1019:     end prune_group;

  1021:     procedure set_server_group_count (
  1022:         customer_id_in in number,
  1023:         group_type_in in number,
  1024:         quantity_in in number,
  1025:                 update_family_countsYN in number := 1
  1026:     ) is
  1027:         group_id number;
  1028:         quantity number;
  1029:     begin
  1030:         quantity := quantity_in;
  1031:         if quantity is not null and quantity < 0 then
  1032:             quantity := 0;

  1034:

  1036:         into    group_id

  1040:             and rsg.group_type = group_type_in;

  1528:     procedure prune_family (

  1544:                 and cfm.channel_id = sc.channel_id
  1545:                 and server_id in (
  1546:                     select    server_id
  1547:                       from    (
  1548:                         select server_id, time, rownum row_number

  1551:                             from    rhnServerChannel         rsc,
  1552:                                     rhnChannelFamilyMembers  rcfm,
  1553:                                     rhnServer                rs
  1554:                             where    1=1
  1555:                                 and rs.org_id = customer_id_in

  1581:         is_fve_in := 'N';

  1591:                                                         update_family_countsYN => 0);
